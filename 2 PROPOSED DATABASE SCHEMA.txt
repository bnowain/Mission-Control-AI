/*
PART 2: PROPOSED DATABASE SCHEMA
Adaptive AI Execution Framework
Chunking + Learning + Telemetry + Codex

Design Goals:
- Hardware agnostic
- Model agnostic
- Portable (SQLite-compatible baseline)
- Efficient on small systems
- Scalable to Postgres + Vector extensions
- Supports telemetry-driven routing
- Supports codex memory + semantic lookup
- Supports chunking for large projects

The agent implementing this may:
- Refactor for normalization
- Add indexes
- Convert to Postgres-specific types
- Add vector storage (pgvector / LanceDB / Chroma)
- Optimize for performance
- Add partitioning for large logs

This schema is a robust baseline.
*/

------------------------------------------------------------
-- CORE ENTITIES
------------------------------------------------------------

CREATE TABLE projects (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    created_at DATETIME NOT NULL,
    config_json TEXT
);

CREATE TABLE models (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    provider TEXT NOT NULL,            -- ollama, openai, anthropic, etc.
    capability_class TEXT NOT NULL,    -- fast_model, reasoning_model, planner_model
    quant TEXT,
    max_context INTEGER,
    created_at DATETIME NOT NULL
);

------------------------------------------------------------
-- TASK DEFINITIONS
------------------------------------------------------------

CREATE TABLE tasks (
    id TEXT PRIMARY KEY,
    project_id TEXT NOT NULL,
    task_type TEXT NOT NULL,          -- bug_fix, refactor_small, architecture_design, etc.
    signature TEXT NOT NULL,          -- hash fingerprint
    status TEXT NOT NULL,             -- pending, running, completed, failed
    created_at DATETIME NOT NULL,
    updated_at DATETIME NOT NULL,
    FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE INDEX idx_tasks_project ON tasks(project_id);
CREATE INDEX idx_tasks_signature ON tasks(signature);

------------------------------------------------------------
-- CHUNKING SYSTEM
------------------------------------------------------------

-- Stores file-level chunks for large codebases

CREATE TABLE file_chunks (
    id TEXT PRIMARY KEY,
    project_id TEXT NOT NULL,
    file_path TEXT NOT NULL,
    chunk_index INTEGER NOT NULL,
    chunk_hash TEXT NOT NULL,
    content TEXT NOT NULL,
    summary TEXT,
    embedding BLOB,                    -- optional vector storage
    created_at DATETIME NOT NULL,
    FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE INDEX idx_chunks_project_file ON file_chunks(project_id, file_path);
CREATE INDEX idx_chunks_hash ON file_chunks(chunk_hash);

-- Optional chunk relationship tracking

CREATE TABLE chunk_dependencies (
    id TEXT PRIMARY KEY,
    parent_chunk_id TEXT NOT NULL,
    dependent_chunk_id TEXT NOT NULL
);

------------------------------------------------------------
-- EXECUTION LOGS (TELEMETRY)
------------------------------------------------------------

CREATE TABLE execution_logs (
    id TEXT PRIMARY KEY,
    task_id TEXT NOT NULL,
    project_id TEXT NOT NULL,
    model_id TEXT NOT NULL,
    context_size INTEGER NOT NULL,
    temperature REAL,
    tokens_generated INTEGER,
    tokens_per_second REAL,
    retries INTEGER,
    score REAL,
    success INTEGER,                    -- 0/1
    human_intervention INTEGER,         -- 0/1
    downstream_impact INTEGER,          -- 0/1
    duration_ms INTEGER,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (task_id) REFERENCES tasks(id),
    FOREIGN KEY (project_id) REFERENCES projects(id),
    FOREIGN KEY (model_id) REFERENCES models(id)
);

CREATE INDEX idx_logs_model ON execution_logs(model_id);
CREATE INDEX idx_logs_task_type ON execution_logs(task_id);

------------------------------------------------------------
-- FAILURE TRACKING
------------------------------------------------------------

CREATE TABLE failure_events (
    id TEXT PRIMARY KEY,
    task_id TEXT NOT NULL,
    error_type TEXT,
    stack_trace_hash TEXT,
    file_path TEXT,
    diff_hash TEXT,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (task_id) REFERENCES tasks(id)
);

CREATE INDEX idx_failure_hash ON failure_events(stack_trace_hash);

------------------------------------------------------------
-- MASTER CODEX (GLOBAL MEMORY)
------------------------------------------------------------

CREATE TABLE master_codex (
    id TEXT PRIMARY KEY,
    issue_signature TEXT NOT NULL,
    category TEXT,
    root_cause TEXT NOT NULL,
    resolution TEXT NOT NULL,
    prevention_guideline TEXT NOT NULL,
    occurrence_count INTEGER DEFAULT 1,
    confidence_score REAL DEFAULT 0.5,
    verified INTEGER DEFAULT 0,
    embedding BLOB,
    first_seen_at DATETIME NOT NULL,
    last_seen_at DATETIME NOT NULL
);

CREATE INDEX idx_master_issue ON master_codex(issue_signature);

------------------------------------------------------------
-- PROJECT-SPECIFIC CODEX
------------------------------------------------------------

CREATE TABLE project_codex (
    id TEXT PRIMARY KEY,
    project_id TEXT NOT NULL,
    issue_signature TEXT NOT NULL,
    module_path TEXT,
    root_cause TEXT NOT NULL,
    resolution TEXT NOT NULL,
    architecture_note TEXT,
    occurrence_count INTEGER DEFAULT 1,
    verified INTEGER DEFAULT 0,
    embedding BLOB,
    first_seen_at DATETIME NOT NULL,
    last_seen_at DATETIME NOT NULL,
    FOREIGN KEY (project_id) REFERENCES projects(id)
);

CREATE INDEX idx_project_codex_project ON project_codex(project_id);

------------------------------------------------------------
-- CODEX PROMOTION CANDIDATES
------------------------------------------------------------

CREATE TABLE codex_candidates (
    id TEXT PRIMARY KEY,
    task_id TEXT NOT NULL,
    issue_signature TEXT NOT NULL,
    proposed_root_cause TEXT,
    proposed_resolution TEXT,
    human_verified INTEGER DEFAULT 0,
    promoted INTEGER DEFAULT 0,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (task_id) REFERENCES tasks(id)
);

------------------------------------------------------------
-- ROUTING PERFORMANCE STATS
------------------------------------------------------------

CREATE TABLE routing_stats (
    id TEXT PRIMARY KEY,
    model_id TEXT NOT NULL,
    task_type TEXT NOT NULL,
    average_score REAL,
    average_retries REAL,
    success_rate REAL,
    sample_size INTEGER,
    last_updated DATETIME NOT NULL,
    FOREIGN KEY (model_id) REFERENCES models(id)
);

CREATE INDEX idx_routing_model_task ON routing_stats(model_id, task_type);

------------------------------------------------------------
-- OPTIONAL EFFICIENCY ADDITIONS
------------------------------------------------------------

-- Hardware capability cache

CREATE TABLE hardware_profiles (
    id TEXT PRIMARY KEY,
    gpu_name TEXT,
    vram_mb INTEGER,
    benchmark_tokens_per_sec REAL,
    created_at DATETIME NOT NULL
);

-- Context escalation logs

CREATE TABLE escalation_events (
    id TEXT PRIMARY KEY,
    task_id TEXT NOT NULL,
    from_context INTEGER,
    to_context INTEGER,
    reason TEXT,
    created_at DATETIME NOT NULL,
    FOREIGN KEY (task_id) REFERENCES tasks(id)
);

------------------------------------------------------------
-- EFFICIENCY IMPROVEMENTS (RECOMMENDED)
------------------------------------------------------------

1. Use SHA-256 hashes for:
   - task_signature
   - stack_trace_hash
   - diff_hash
   - chunk_hash

2. Add vector index support:
   - If Postgres → use pgvector
   - If SQLite → external vector store
   - Enable semantic issue matching

3. Periodic aggregation job:
   - Update routing_stats
   - Decay old performance
   - Detect model drift

4. Index heavy-use fields:
   - task_type
   - model_id
   - issue_signature

5. Partition execution_logs by project if scaling.

6. Add TTL cleanup policy for old logs.

------------------------------------------------------------
ARCHITECTURAL NOTES
------------------------------------------------------------

- file_chunks supports semantic retrieval and context injection.
- master_codex prevents global mistake repetition.
- project_codex captures architecture quirks.
- codex_candidates prevents overfitting bad lessons.
- routing_stats enables adaptive model switching.
- failure_events supports clustering.
- escalation_events supports context optimization tuning.

IMPORTANT DESIGN PRINCIPLE
------------------------------------------------------------

The database must:
- Never depend on a specific model
- Never depend on specific GPU
- Store deterministic evidence only
- Allow full replay of execution decisions

This schema is a starting blueprint.
The implementing agent may refactor for normalization,
performance, or improved modularity.
